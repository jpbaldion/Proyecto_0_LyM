defProc putCB (c , b)
{
    drop (c) ;
    letGo (b);
    walk (1) ;
    putCB (1 ,1)
}

{
    jump (3 ,3) ;
    putCB (2 ,1)
}

defProc goNorth ()
{

    while can ( walk (1 , north )) { walk (1 , north ) };
    putCB (1 ,1)

}

{
jump (3 ,4) ;
putCB (5 ,5) ;
goNorth ()
}

defProc goWest ()
{

    if can ( walk (1 , west ) ) { walk (1 , west )} else { nop () };
    goNorth () ;
    goWest () ;

}

{
jump (4 ,5) ;
putCB (6 ,7) ;
goNorth () ;
goWest () ;
goNorth ()
}
